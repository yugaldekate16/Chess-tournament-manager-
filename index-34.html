Enterprise Auth & Recovery System
Production-ready, zero-backdoor authentication using Firebase Auth, Firestore, and server-side session management. Email/password, Google, Chess.com login, MFA, role-based access, and Super Admin recovery controls.
🚫
Zero-backdoor guarantee: No hidden URLs, hardcoded passwords, secret bypass logins, or localStorage authentication anywhere in this system.
🔐 Firebase Auth IDENTITY
Email/Password, Google OAuth, Chess.com-linked identity. Server-side session cookies replace localStorage entirely. ID tokens verified by Firebase Admin SDK on every request.
🛡 Role-Based Access RBAC
Four roles enforced by Firestore security rules AND backend middleware: Super Admin, Admin, Organizer, Player. Every route is guarded server-side, not just client-side.
♟ Chess.com Integration API
Auto-creates player accounts on first Chess.com login. Deduplicates by chessComId then chessComUsername. Syncs avatar, rating, and profile data from the public PubAPI.
🔄 Recovery System RECOVERY
Multi-step identity verification, MFA reset, recovery codes, and Super Admin override path. Every recovery action is logged in auditLogs with full context.
📋 Audit Logging COMPLIANCE
Every auth event writes an immutable audit record: actorUserId, action, targetUserId, timestamp, IP, device fingerprint, and metadata. Tamper-evident by Firestore rules.
🍪 Secure Sessions SESSION
httpOnly + secure + sameSite=strict cookies. Session state lives in Firestore, not the browser. Revocation is instant. Token refresh handled server-side transparently.

Authentication
Firebase Auth Setup
Client sends ID token to server; server creates httpOnly session cookie. No token ever stored in localStorage.
client/js/session.js — Token Exchange
session.jsJS
// After Firebase signs in, exchange the ID token for a
// server-managed httpOnly session cookie. Never touch localStorage.
export async function exchangeTokenForSession(rememberMe = false) {
  const user  = auth.currentUser;
  if (!user) throw new Error('No authenticated user');

  // forceRefresh=true ensures token is fresh
  const idToken = await user.getIdToken(true);

  const res = await fetch('/api/auth/session', {
    method:  'POST',
    headers: { 'Content-Type': 'application/json',
               'X-CSRF-Token': getCsrfToken() },
    body: JSON.stringify({ idToken, rememberMe }),
    credentials: 'include'   // sends/receives httpOnly cookie
  });

  if (!res.ok) throw new Error('Session creation failed');
  return res.json();
}

// Sign out: revoke server session, then clear Firebase local state
export async function secureLogout() {
  await fetch('/api/auth/logout', {
    method: 'POST',
    headers: { 'X-CSRF-Token': getCsrfToken() },
    credentials: 'include'
  });
  await auth.signOut();
  window.location.href = '/login.html';
}

// POST /api/auth/session — verify ID token, create session cookie
router.post('/session', csrfProtect, async (req, res) => {
  const { idToken, rememberMe } = req.body;
  try {
    // 1. Verify token is fresh (issued within last 5 minutes)
    const decoded = await admin.auth().verifyIdToken(idToken);
    if (Date.now() / 1000 - decoded.iat > 300)
      return res.status(401).json({ error: 'Token too old' });

    // 2. Check account status in Firestore
    const userDoc  = await db.collection('users').doc(decoded.uid).get();
    const userData = userDoc.data();
    if (!userData || userData.status === 'disabled')
      return res.status(403).json({ error: 'Account disabled' });

    // 3. Check MFA required for admin/super_admin
    if (['admin','super_admin'].includes(userData.role) && userData.mfaEnabled)
      if (!decoded.firebase.sign_in_second_factor)
        return res.status(403).json({ error: 'MFA_REQUIRED' });

    // 4. Create Firebase session cookie
    const expiresIn = rememberMe ? 14*24*60*60*1000 : 3600000;
    const cookie    = await admin.auth().createSessionCookie(idToken, { expiresIn });

    // 5-7. Store session in Firestore, update lastLogin, write audit log
    await sessionRef.set({ uid: decoded.uid, active: true, ip: req.ip, ... });
    await createAuditLog({ actorUserId: decoded.uid, action: 'LOGIN_SUCCESS', ... });

    // 8. Set cookie — httpOnly, secure, sameSite strict
    res.cookie('__session', cookie, {
      httpOnly: true, secure: true, sameSite: 'strict', maxAge: expiresIn
    });
    res.json({ role: userData.role, uid: decoded.uid });
  } catch (err) { res.status(401).json({ error: 'Authentication failed' }); }
});

// POST /api/auth/logout — revoke session cookie + refresh tokens
router.post('/logout', csrfProtect, verifySession, async (req, res) => {
  await admin.auth().revokeRefreshTokens(req.user.uid);
  await db.collection('sessions').doc(req.user.sessionId).update({ active: false });
  res.clearCookie('__session');
  res.json({ ok: true });
});
Session Management
Server-Side Sessions
All session state lives in Firestore. Browser only holds an httpOnly cookie it cannot read via JS.
✅
Cookies are httpOnly (no JS access), secure (HTTPS only), sameSite=strict (CSRF defense). Firestore records enable instant server-side revocation.
server/middleware/verifySession.js
verifySession.jsJS
module.exports = async function verifySession(req, res, next) {
  const sessionCookie = req.cookies?.['__session'];
  if (!sessionCookie) return res.status(401).json({ error: 'Unauthorized' });
  try {
    // Verify the Firebase session cookie (checkRevoked=true)
    const decoded = await admin.auth().verifySessionCookie(sessionCookie, true);

    // Cross-check active session record in Firestore
    const sessDoc = await db.collection('sessions')
      .where('uid', '==', decoded.uid).where('active', '==', true).limit(1).get();
    if (sessDoc.empty) throw new Error('Session revoked');

    const user = (await db.collection('users').doc(decoded.uid).get()).data();
    if (!user || user.status === 'disabled') throw new Error('Account disabled');

    req.user = { uid: decoded.uid, role: user.role,
                 sessionId: sessDoc.docs[0].id, ...user };
    next();
  } catch {
    res.clearCookie('__session');
    res.status(401).json({ error: 'Session invalid or expired' });
  }
};

// Usage: router.get('/admin', verifySession, requireRole('admin','super_admin'), handler)
module.exports = function requireRole(...allowedRoles) {
  return (req, res, next) => {
    const { role } = req.user;
    if (role === 'super_admin' || allowedRoles.includes(role)) return next();
    return res.status(403).json({ error: 'Insufficient permissions' });
  };
};
Rate Limiting — server/middleware/rateLimit.js
rateLimit.jsJS
const MAX_ATTEMPTS = 5;
const LOCK_MINUTES = 30;

module.exports.checkLoginRateLimit = async function(email, ip) {
  const key = `${email}:${ip}`.toLowerCase();
  const doc = await db.collection('loginAttempts').doc(key).get();
  if (doc.exists) {
    const { attempts, lockedUntil } = doc.data();
    if (lockedUntil && lockedUntil.toDate() > new Date())
      throw new Error(`RATE_LIMIT:${Math.ceil((lockedUntil.toDate()-Date.now())/60000)}`);
    if (attempts >= MAX_ATTEMPTS)
      await ref.update({ lockedUntil: new Date(Date.now()+LOCK_MINUTES*60000) });
  }
};

server/utils/crypto.js — Recovery Code Generation
crypto.jsJS
const crypto = require('crypto');
const bcrypt = require('bcrypt');

// Generate 8 cryptographically secure recovery codes
async function generateRecoveryCodes() {
  const codes = Array.from({ length: 8 }, () =>
    crypto.randomBytes(5).toString('hex').toUpperCase()
      .match(/.{1,4}/g).join('-')   // format: ABCD-1234-EF56
  );
  const hashed = await Promise.all(codes.map(c => bcrypt.hash(c, 12)));
  return { codes, hashed }; // show codes[] ONCE, store hashed[]
}

async function verifyRecoveryCode(submitted, hashes) {
  for (const [i, hash] of hashes.entries())
    if (await bcrypt.compare(submitted, hash)) return { valid: true, index: i };
  return { valid: false };
}

Chess.com Integration
Chess.com Login & Auto-Add
On Chess.com login, the system deduplicates by chessComId first, then chessComUsername. New players auto-created with role=player.
1
Player enters Chess.com username
Client fetches public profile: api.chess.com/pub/player/{username}
2
Server receives chessComId + username
Route POST /api/chess/login
3
Search Firestore by chessComId
If found → update record, sync profile data → return session
4
If not found, search by chessComUsername
If found → link chessComId, update record → return session
5
No existing record → auto-create player
New Firestore doc with role=player, status=active. Writes CREATE audit log.
6
Session issued
Firebase custom token minted, client exchanges for session cookie
server/services/chessService.js
chessService.jsJS
async function chessComLogin({ chessComUsername, requestIp, userAgent }) {
  // 1. Fetch public profile from Chess.com PubAPI
  const pubRes  = await fetch(`https://api.chess.com/pub/player/${chessComUsername.toLowerCase()}`);
  if (!pubRes.ok) throw new Error('Chess.com player not found');
  const profile = await pubRes.json();
  const chessComId     = String(profile.player_id);
  const normalizedUser = profile.username.toLowerCase();

  // 2. Search by chessComId first
  let snap = await db.collection('users')
    .where('chessComId', '==', chessComId).limit(1).get();

  // 3. Fallback: search by username
  if (snap.empty) snap = await db.collection('users')
    .where('chessComUsername', '==', normalizedUser).limit(1).get();

  let uid, action;
  if (!snap.empty) {
    // 4a. Found — update existing record
    uid = snap.docs[0].id;
    await snap.docs[0].ref.update({
      chessComId, chessComUsername: normalizedUser,
      lastLogin: admin.firestore.FieldValue.serverTimestamp()
    });
    action = 'CHESS_PLAYER_MATCH_FOUND';
  } else {
    // 4b. Not found — auto-create new player
    const newRef = db.collection('users').doc();
    uid = newRef.id;
    await newRef.set({
      uid, username: profile.username, email: null,
      chessComId, chessComUsername: normalizedUser,
      role: 'player', status: 'active', mfaEnabled: false,
      createdAt: admin.firestore.FieldValue.serverTimestamp(),
      lastLogin: admin.firestore.FieldValue.serverTimestamp()
    });
    await admin.auth().createUser({ uid, displayName: profile.username });
    action = 'CHESS_PLAYER_AUTO_CREATED';
  }

  await createAuditLog({ actorUserId: uid, action, ip: requestIp });
  const customToken = await admin.auth().createCustomToken(uid, { role: 'player' });
  return { customToken, uid };
}

firestore/rules.firestore
rules.firestoreFIRESTORE
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function getRole() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role;
    }
    function isSuper()  { return getRole() == 'super_admin'; }
    function isAdmin()  { return getRole() in ['admin', 'super_admin']; }

    match /users/{uid} {
      allow read: if isAdmin() || request.auth.uid == uid;
      allow create: if isSuper();
      allow update: if isSuper() ||
        (request.auth.uid == uid && !request.resource.data.diff(resource.data)
          .affectedKeys().hasAny(['role','status','mfaEnabled','recoveryCodes']));
      allow delete: if isSuper();
    }

    match /auditLogs/{logId} {
      allow read: if isAdmin();
      allow create: if request.auth != null;
      allow update, delete: if false;   // IMMUTABLE — never editable
    }

    match /sessions/{sessionId} {
      allow read, write: if false;      // server-only via Admin SDK
    }
  }
}
